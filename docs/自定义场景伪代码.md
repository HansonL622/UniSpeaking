自定义场景服务体系
│
├── CustomSceneController
│   │
│   ├── [调用] CustomSceneService
│   │   ├── create(userId, command)
│   │   ├── getById(sceneId)
│   │   ├── update(sceneId, command)
│   │   └── delete(sceneId)
│   │
│   └── [调用] SceneContentService
│       ├── generate(customScene, profile)
│       └── getOrGenerate(sceneId, profile)
│
├── CustomSceneSessionController
│   │
│   └── [调用] CustomSceneSessionService
│       │
│       ├── [继承] SessionService
│       ├── [实现复用] AbstractSceneSessionService
│       ├── start(StartCommand)
│       ├── pause(localSessionId)
│       ├── resume(localSessionId)
│       ├── interrupt(localSessionId)
│       └── complete(localSessionId)
│       │
│       ├── [调用] AuthService
│       ├── [调用] ProfileService
│       ├── [调用] CustomSceneService
│       ├── [调用] SceneContentService
│       ├── [调用] CustomScenePromptService
│       ├── [调用] LearningFlowService
│       ├── [调用] RealtimeConnectionService
│       └── [调用] SessionStateService
│
├── LearningFlowService
│   ├── createProgress(localSessionId, content)
│   ├── getCurrentStage(localSessionId)
│   ├── submitPractice(localSessionId, practice)
│   └── moveToNextStage(localSessionId)
│       │
│       ├── [调用] PracticeAudioService
│       ├── [调用] EvaluationService
│       ├── [调用] LearningRecordService
│       └── [调用] MistakeBookService
│
├── CustomSceneRealtimeEventService
│   ├── session.created
│   │   └── CustomSceneSessionService.bindProviderSession()
│   ├── session.updated
│   │   └── CustomSceneSessionService.activate()
│   ├── transcript.completed
│   │   └── CustomSceneConversationService.appendMessage()
│   └── error
│       └── CustomSceneSessionService.fail()
│
├── CustomSceneConversationService
│   └── 当前在进程内存中暂存对话文字
│
├── LearningRecordService
│   └── 当前在进程内存中暂存学习进度和评分
│
├── MistakeBookService
│   └── 当前在进程内存中暂存错词、错句
│
├── SessionMemoryService
│   └── 会话完成时：对话摘要 + 历史 memory → 更新用户档案
│
└── LearningReportService
    └── 汇总评分、错题和学习记录，生成报告

// 1. 用户创建自定义场景
CustomScene customScene =
    CustomSceneService.create(userId, createCommand);


// 2. 根据主题和用户画像生成学习内容
UserProfile profile = ProfileService.getProfile(userId);

SceneLearningContent content =
    SceneContentService.generate(customScene, profile);


// 3. 启动自定义场景会话
StartResult result =
    CustomSceneSessionService.start(
        StartCommand.builder()
            .setSceneId(customScene.getSceneId())
            .setOfferSdp(offerSdp)
            .build()
    );


// 4. 启动时准备场景和学习流程
SceneConfig sceneConfig =
    CustomSceneService.getConfig();

CustomScene customScene =
    CustomSceneService.getById(sceneId);

SceneLearningContent content =
    SceneContentService.getOrGenerate(
        sceneId,
        profile
    );

LearningProgress progress =
    LearningFlowService.createProgress(
        localSessionId,
        content
    );

SessionPrompt prompt =
    CustomScenePromptService.build(
        profile,
        sceneConfig,
        customScene,
        content,
        progress.getCurrentStage()
    );


// 5. 服务端读取永久 API Key，代理完成 SDP 交换
AnswerSdp answerSdp =
    RealtimeConnectionService.connect(
        sceneConfig.getProviderType(),
        offerSdp
    );


// 6. 跟读练习和评分
EvaluationResult evaluationResult =
    LearningFlowService.submitPractice(
        localSessionId,
        practiceAudio
    );

// 内部调用：
// PracticeAudioService
// EvaluationService
// LearningRecordService
// MistakeBookService


// 7. 达到分数后进入下一阶段
LearningFlowService.moveToNextStage(
    localSessionId
);


// 8. 对话文字暂存在当前进程内存
CustomSceneRealtimeEventService.handleEvent(
    localSessionId,
    realtimeEvent
);

CustomSceneConversationService.appendMessage(
    localSessionId,
    conversationMessage
);


// 9. 完成场景并生成报告
CustomSceneSessionService.complete(
    localSessionId
);

// complete() 同时生成本次会话摘要，
// 与用户档案中的历史 memory 合并后写回内存档案。

LearningReport report =
    LearningReportService.generateReport(
        localSessionId
    );
