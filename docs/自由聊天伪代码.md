@Service
public class FreeChatSessionService extends SessionService {

    @Override
    public StartResult start(StartCommand command) {

        // 1. 获取当前用户和用户画像
        String userId =
            AuthService.getCurrentUserId();

        UserProfile profile =
            ProfileService.getProfile(userId);


        // 2. 获取自由聊天配置并校验权限
        SceneConfig sceneConfig =
            FreeChatSceneService.getConfig();

        FreeChatSceneService.validateAccess(userId);

        UsageQuotaService.validate(userId);


        // 3. 创建本地自由聊天会话
        String localSessionId =
            SessionIdGenerator.generate();

        FreeChatSceneSession session =
            new FreeChatSceneSession(
                localSessionId,
                userId
            );


        // 4. 生成 Prompt，并设置会话配置
        SessionPrompt prompt =
            FreeChatPromptService.build(
                profile,
                sceneConfig
            );

        session.setSceneId(sceneConfig.getSceneId());
        session.setPrompt(prompt);
        session.setProviderType(
            sceneConfig.getProviderType()
        );

        session.setVoiceId(
            profile.getVoiceId() != null
                ? profile.getVoiceId()
                : sceneConfig.getDefaultVoiceId()
        );


        // 5. 预占用户额度并暂存会话
        UsageQuotaService.reserve(
            userId,
            localSessionId
        );

        SessionStateService.create(session);


        // 6. 服务端代理完成 SDP 交换
        RealtimeConnectionResult connection =
            RealtimeConnectionService.connect(
                session.getProviderType(),
                command.getOfferSdp()
            );


        // 7. 更新为等待模型会话绑定状态
        session.markConnecting();

        SessionStateService.update(session);


        // 8. 启动 providerSessionId 绑定超时检测
        SessionBindingTimeoutService.schedule(
            localSessionId,
            BIND_TIMEOUT
        );


        // 9. 返回前端建连信息
        return StartResult.builder()
            .setLocalSessionId(localSessionId)
            .setAnswerSdp(connection.getAnswerSdp())
            .setSystemPrompt(prompt)
            .setVoiceId(session.getVoiceId())
            .build();
    }


    @Override
    public void handleEvent(
            String localSessionId,
            SessionEvent event) {

        FreeChatSceneSession session =
            SessionStateService.get(localSessionId);

        switch (event.getType()) {

            // 模型返回 session.created
            case PROVIDER_SESSION_CREATED -> {

                session.bindProviderSession(
                    event.getProviderSessionId()
                );

                SessionBindingTimeoutService.cancel(
                    localSessionId
                );

                UsageQuotaService.startMetering(
                    localSessionId
                );

                SessionStateService.update(session);
            }


            // 模型完成 Prompt、音色等配置
            case PROVIDER_SESSION_UPDATED -> {
                session.activate();
                SessionStateService.update(session);
            }


            // 用户或 AI 的完整字幕
            case USER_TRANSCRIPT_COMPLETED,
                 AI_TRANSCRIPT_COMPLETED -> {
                FreeChatConversationService.appendMessage(
                    localSessionId,
                    event.toConversationMessage()
                );
            }


            // 用户暂停会话
            case SESSION_PAUSE -> {

                session.pause();

                SessionStateService.update(session);
            }


            // 用户恢复会话
            case SESSION_RESUME -> {
                session.resume();
                SessionStateService.update(session);
            }


            // 用户打断 AI
            case SESSION_INTERRUPT -> {

                session.recordInterrupt();

                SessionStateService.update(session);
            }


            // 用户正常结束会话
            case SESSION_COMPLETE -> {
                session.complete();
                SessionStateService.update(session);
                UsageQuotaService.settle(
                    localSessionId
                );
                SessionMemoryService.updateMemory(
                    localSessionId
                );
            }


            // 模型或连接发生异常
            case SESSION_ERROR -> {
                session.fail(
                    event.getErrorCode(),
                    event.getErrorMessage()
                );
                SessionStateService.update(session);
                UsageQuotaService.settle(
                    localSessionId
                );
            }


            // providerSessionId 绑定超时
            case BIND_TIMEOUT -> {
                if (session.getProviderSessionId() == null) {
                    session.fail(
                        "PROVIDER_BIND_TIMEOUT",
                        "模型会话绑定超时"
                    );
                    SessionStateService.update(session);
                    UsageQuotaService
                        .settleReservedQuota(
                            localSessionId
                        );
                    SessionWebSocketHandler
                        .sendCloseWebRtc(
                            localSessionId
                        );
                }
            }
        }
    }
}
